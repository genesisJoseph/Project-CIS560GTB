CREATE DATABASE FarmOfEggs;
GO

USE FarmOfEggs;
GO

-- 1. ChickenBreed Table (No dependencies)
CREATE TABLE ChickenBreed (
    ChickenBreedID INT PRIMARY KEY IDENTITY(1,1),
    BreedName NVARCHAR(100) UNIQUE NOT NULL,
    Species NVARCHAR(100) NOT NULL
);

-- 2. Farm Table (No dependencies)
CREATE TABLE Farm (
    FarmID INT PRIMARY KEY IDENTITY(1,1),
    Location NVARCHAR(255) NOT NULL,
    Name NVARCHAR(100) UNIQUE NOT NULL,
    PhoneNumber NVARCHAR(20) UNIQUE NOT NULL,
    Owner NVARCHAR(100) NOT NULL
);

-- 3. Retail Table (No dependencies)
CREATE TABLE Retail (
    RetailID INT PRIMARY KEY IDENTITY(1,1),
    PhoneNumber NVARCHAR(20) UNIQUE NOT NULL,
    Name NVARCHAR(100) UNIQUE NOT NULL,
    RetailFee DECIMAL(10, 2) NOT NULL
);

-- 4. ChickenGroup Table (Depends on Farm and ChickenBreed)
CREATE TABLE ChickenGroup (
    ChickenGroupID INT PRIMARY KEY IDENTITY(1,1),
    Confinement NVARCHAR(100),
    FarmID INT NOT NULL,
    ChickenBreedID INT NOT NULL,
    FOREIGN KEY (FarmID) REFERENCES Farm(FarmID),
    FOREIGN KEY (ChickenBreedID) REFERENCES ChickenBreed(ChickenBreedID)
);

-- 5. BatchEgg Table (Depends on ChickenGroup)
CREATE TABLE BatchEgg (
    BatchEggID INT PRIMARY KEY IDENTITY(1,1),
    ChickenGroupID INT NOT NULL,
    Size NVARCHAR(50),
    DateProduced DATE NOT NULL,
    BestByDate DATE,
    Grade NVARCHAR(10),
    EggColor NVARCHAR(50),
    FOREIGN KEY (ChickenGroupID) REFERENCES ChickenGroup(ChickenGroupID)
);

-- 6. EggSales Table (Depends on BatchEgg and Retail)
CREATE TABLE EggSales (
    EggSalesID INT PRIMARY KEY IDENTITY(1,1),
    UnitPrice DECIMAL(10, 2) NOT NULL,
    AvgUnit INT,
    BatchID INT NOT NULL,
    RetailID INT NOT NULL,
    CanceledOn DATETIME NULL,
    FOREIGN KEY (BatchID) REFERENCES BatchEgg(BatchEggID),
    FOREIGN KEY (RetailID) REFERENCES Retail(RetailID)
);
